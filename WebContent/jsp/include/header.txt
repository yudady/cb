<script type="text/javascript">

	/**
	 * cb 命名空間
	 */
	var cb = function(){
		
		function baseUrl(path){
			path = path || "";
			return "" + '<c:url value="/"/>' + path;
		}
		

		function dialog(message) {
			var setting = message || "dialog need text message";
				
			$("#dialog").html("<br/><br/><br/><div>" + setting + "</div><br/>").dialog({
				modal: true,
				buttons : {
					"OK" : function() {
						$(this).dialog("close");
					}
				}
			});
			$(".ui-dialog-titlebar").hide();
		}

		//=============		
		return {
			getSafeUrl : function(path) {
				return baseUrl(path);
			},
			openAlertDialog : function(message) {
				return dialog(message);
			},
			isBidderLogin : function() {
				if(! $("#loginOutFormBtn").is(":visible")){
					return false;
				}
				return true;
			}
		};
	}();
</script>

<script type="text/javascript">
$(function() {
	$("#loginBtn").click(function(){
		
		$("#loginForm").dialog({ 
			resizable: false,
			position: { of: $( "#loginBtn" ) , my: "center top", at: "left bottom"  },
			draggable: false,
			height: 300,
			width:400,
			title: "Account Login"
		
		});
	});
	
	$("#loginFormBtn").on('click',function(){
		$("#url").val(window.location.href);
		var form = $("#loginForm");
		var url = form.attr('action');
		$.ajax({
			type : "POST",
			url : url,
			data : form.serialize(), 
			success : function(data) {
				if(data.success){
					window.location.reload(true);
				}else{
					cb.openAlertDialog(data.msg);
				}
			}
		});
		return false;
	});

	$("#loginOutFormBtn").on('click', function() {
		$("#url").val(window.location.href);
		var form = $("#loginOutForm");
		var url = form.attr('action');
		$.ajax({
			type : "POST",
			url : url,
			data : form.serialize(), 
			success : function(data) {
				if(data.success){
					window.location.reload(true);
				}else{
					cb.openAlertDialog(data.msg);
				}
			}
		});
		return false;
	});

	$("#emailsubscribe input.subscribeemails").on('focusin', function() {
		$(this).val('');
	});
	$("#searchItem a").on('click', function(event) {
		$("#searchItem").submit();
		event.preventDefault();
	});

});
</script>
